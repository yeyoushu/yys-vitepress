let confirm = (fromParam: FromInstance | Ref<FromInstance | null> | null) => {
  // 支持 slot 传入的 ref 或直接实例
  const from =
    fromParam && (fromParam as any).value !== undefined
      ? (fromParam as any).value
      : fromParam;

  if (!from) {
    console.warn("confirm: from is null or undefined, cannot validate");
    return;
  }

  try {
    let handled = false;

    const handleResult = (valid: boolean) => {
      if (valid) {
        models.value = from.getFieldsValue();
        ElMessage({ message: "验证通过", type: "success" });
      } else {
        ElMessageBox.alert("表单填写不完整", "提示", {
          confirmButtonText: "确定",
          type: "warning",
        });
      }
    };

    // 以回调方式调用 validate（类型安全），并取得可能返回的 Promise
    const maybePromise = from.validate(
      (valid: boolean, _invalidFields?: any) => {
        handled = true;
        handleResult(!!valid);
      }
    ) as void | Promise<boolean>;

    // 如果返回 Promise 且回调尚未执行，则使用 Promise 结果
    if (maybePromise && typeof (maybePromise as any).then === "function") {
      (maybePromise as Promise<boolean>)
        .then((valid) => {
          if (!handled) handleResult(!!valid);
        })
        .catch((err) => {
          console.error("validate error", err);
          ElMessageBox.alert("表单验证失败", "提示", {
            confirmButtonText: "确定",
            type: "warning",
          });
        });
    }
  } catch (err) {
    console.error("validate error", err);
    ElMessageBox.alert("表单验证失败", "提示", {
      confirmButtonText: "确定",
      type: "warning",
    });
  }
};

表单事件 #

事件名说明参数
onPreview点击文件列表中已上传的文件时的钩子file: UploadFile
onRemove文件列表移除文件时的钩子file: UploadFile, fileList: UploadFiles
onSuccess文件上传成功时的钩子response: any, file: UploadFile, fileList: UploadFiles
onError文件上传失败时的钩子err: any, file: UploadFile, fileList: UploadFiles
onProgress文件上传时的钩子event: any, file: UploadFile, fileList: UploadFiles
onChange文件状态改变时的钩子，添加文件、上传成功和上传失败时都会被调用uploadFile: UploadFile, uploadFiles: UploadFiles
onExceed当超出限制时，执行的钩子函数files: File[], uploadFiles: UploadUserFile[]
beforeUpload上传文件之前的钩子，参数为上传的文件， 若返回false或者返回 Promise 且被 reject，则停止上传。rawFile: UploadRawFile
beforeRemove删除文件之前的钩子，参数为上传的文件和文件列表， 若返回 false 或者返回 Promise 且被 reject，则停止删除。uploadFile: UploadFile,uploadFiles: UploadFiles
