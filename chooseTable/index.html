表格 #

可配置型表格，支持分页与行选择。

按需引入 #


import chooseTable from 'yys-element-components/table'
import 'yys-element-components/table/style.css'
// min.ts中注册
app.use(chooseTable)

基本用法 #

<script setup lang="ts">
import type { TableOptions } from "@/components/table/type";
import { ref, onMounted } from "vue";
import { ElMessage } from "element-plus";
import axios from "axios";
interface DataItem {
  date: string;
  name: string;
  address: string;
}
// 表示点击哪个操作按钮允许编辑行（使用每行 key）
let editRowIndex = ref<string>("");
// 每行的备份数据（用于取消恢复）
const rowBackups = new Map<string, any>();
let tableData = ref<DataItem[]>([]);
//  定时器
// setTimeout(() => {
//   tableData.value = [
//     {
//       date: "2016-05-03",
//       name: "Tom",
//       address: "No. 189, Grove St, Los Angeles",
//     },
//     {
//       date: "2016-05-02",
//       name: "aJiao",
//       address: "No. 189, Grove St, Los Angeles",
//     },
//     {
//       date: "2016-05-04",
//       name: "Tom",
//       address: "No. 189, Grove St, Los Angeles",
//     },
//     {
//       date: "2016-05-01",
//       name: "Tom",
//       address: "No. 189, Grove St, Los Angeles",
//     },
//   ];
// }, 2000);

// 表格配置
let Options: TableOptions[] = [
  {
    label: "日期",
    prop: "date",
    width: "240",
    slot: "date",
    align: "center",
    editable: true,
  },
  {
    label: "姓名",
    prop: "name",
    width: "200",
    slot: "name",
    align: "center",
  },
  {
    label: "地址",
    prop: "address",
    width: "300",
    align: "center",
    editable: true,
  },

  {
    label: "操作",
    prop: "actions",
    width: "200",
    align: "center",
    actions: true,
  },
];

let current = ref<number>(1);
let pageSize = ref<number>(10);
let total = ref<number>(0);

// 统一数据加载函数：初始与翻页均使用
const loadData = async (page = current.value, size = pageSize.value) => {
  try {
    const res = await axios.post("/api/list", {
      current: page,
      pageSize: size,
    });
    console.log("res后端返回数据", res);

    // 后端返回结构：{ code, message, data: { current, pageSize, rows, total, totalPages } }
    tableData.value = res.data.data.rows || [];
    total.value = res.data.data.total || 0;
  } catch (e) {
    ElMessage.error("加载表格数据失败");
    tableData.value = [];
    total.value = 0;
  }
};

onMounted(() => {
  loadData(current.value, pageSize.value);
});

// 翻页回调（chooseTable 会触发）
const onPageChange = (payload: { current: number; pageSize: number }) => {
  if (!payload) return;
  current.value = payload.current ?? current.value;
  pageSize.value = payload.pageSize ?? pageSize.value;
  loadData(current.value, pageSize.value);
};
// 编辑
let Edit = (scope: any) => {
  // 使用行索引作为 key（也可以使用某个唯一字段）
  const key = String(scope.$index);
  // 备份当前行数据（深拷贝）
  rowBackups.set(key, JSON.parse(JSON.stringify(scope.row)));
  editRowIndex.value = key;
  console.log("编辑", scope);
};
// 删除（退出编辑并清理备份）
let Delete = (scope: any) => {
  // const key = String(scope.$index);
  // // 清理备份并退出编辑状态
  // if (rowBackups.has(key)) rowBackups.delete(key);
  editRowIndex.value = "";
  console.log("删除", scope);
};
// 确认编辑：提交并清理备份
let confirm = (scope: any) => {
  const key = String(scope.$index);
  if (rowBackups.has(key)) rowBackups.delete(key);
  // 关闭编辑状态
  editRowIndex.value = "";
  console.log("点击了编辑确认", scope);
};
// 取消编辑：恢复备份并清理
let cancel = (scope: any) => {
  const key = String(scope.$index);
  const backup = rowBackups.get(key);
  if (backup) {
    // 恢复整行数据（保持响应性）
    tableData.value.splice(scope.$index, 1, backup);
    rowBackups.delete(key);
  }
  // 关闭编辑状态
  editRowIndex.value = "";
  console.log("点击了编辑取消", scope);
};
</script>

props #

属性说明类型可选值默认值
options表格配置选项TableOptions[]-[]
data表格数据any[]-[]
elementLoadingText加载文案String-加载中...
elementLoadingSpinner加载图标名String--
elementLoadingBackground加载背景色String--

editRowIndex编辑行的标识String--

currentPage当前是第几页的数据Number-1
pageSize每页多少条数据Number-10
total数据总数Number-100
size分页大小String""/ default / small/ largesmall
hideOnSinglePage是否显示分页booleantrue/falsefalse

FromOptions数据格式说明 #

属性说明类型
label表头string
prop字段名称string
width列宽度string / number
align对齐方式'left' / 'center'/ 'right'
slot自定义模板string
actions是否代表操作项boolean
editable是否可编辑单元格boolean

表单格事件 #

事件名说明参数
confirm可编辑单元格确认事件scope: 当前行数据
cancel可编辑单元格取消事件scope: 当前行数据
page-change统一分页变更事件payload: current, pageSize
editRowIndex可编辑单元格取消事件scope: 当前行数据
update:currentPage当前页码分页数据
update:pageSize每页显示数量分页数据
